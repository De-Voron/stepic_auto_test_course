train_df['Fare_no_out'] = train_df['Fare']
fare_pclass_1 = train_df[train_df['Pclass'] == 1]['Fare']
fare_pclass_2 = train_df[train_df['Pclass'] == 2]['Fare']
fare_pclass_3 = train_df[train_df['Pclass'] == 3]['Fare']
fare_pclass_1_no_out = fare_pclass_1[(fare_pclass_1 - fare_pclass_1.mean()).abs()  
                                   < 2 * fare_pclass_1.std()]
fare_pclass_2_no_out = fare_pclass_2[(fare_pclass_2 - fare_pclass_2.mean()).abs()  
                                   < 2 * fare_pclass_2.std()]
fare_pclass_3_no_out = fare_pclass_3[(fare_pclass_3 - fare_pclass_3.mean()).abs() 
                                   < 2 * fare_pclass_3.std()]
train_df['Fare_no_out'] = fare_pclass_1_no_out.append(fare_pclass_2_no_out)\
                                             .append(fare_pclass_3_no_out)

sns.boxplot(x='Pclass', y='Fare_no_out', data=train_df);
